// 初始化 Isotope
const grid1 = new Isotope('#grid1', {
  itemSelector: '.grid-item',
  layoutMode: 'masonry'
});

// 按钮点击过滤
const filterButtons = document.querySelectorAll('#grid1 ~ .controls button');
filterButtons.forEach(btn => {
  btn.addEventListener('click', () => {
    // 移除所有激活状态
    filterButtons.forEach(b => b.classList.remove('is-checked'));
    // 激活当前按钮
    btn.classList.add('is-checked');
    // 应用过滤
    grid1.arrange({ filter: btn.getAttribute('data-filter') });
  });
});

// 存储每组过滤条件
const filters = {};

// 组合过滤逻辑
const buttonGroups = document.querySelectorAll('#grid3 ~ .controls .button-group');
buttonGroups.forEach(group => {
  const filterGroup = group.getAttribute('data-filter-group');
  const buttons = group.querySelectorAll('button');
  
  buttons.forEach(btn => {
    btn.addEventListener('click', () => {
      buttons.forEach(b => b.classList.remove('is-checked'));
      btn.classList.add('is-checked');
      
      // 存储该组过滤条件
      filters[filterGroup] = btn.getAttribute('data-filter');
      
      // 合并所有过滤条件
      const filterValue = Object.values(filters).join('');
      grid3.arrange({ filter: filterValue || '*' });
    });
  });
});

// 切换布局模式
const layoutButtons = document.querySelectorAll('#grid4 ~ .controls button');
layoutButtons.forEach(btn => {
  btn.addEventListener('click', () => {
    layoutButtons.forEach(b => b.classList.remove('is-checked'));
    btn.classList.add('is-checked');
    
    const layoutMode = btn.getAttribute('data-layout');
    const options = { layoutMode: layoutMode };
    
    // 为不同布局模式设置特定参数
    if (layoutMode === 'cellsByRow') {
      options.cellsByRow = { columnWidth: 200, rowHeight: 200 };
    }
    
    grid4.arrange(options);
  });
});

// 从 URL Hash 获取过滤条件
function getHashFilter() {
  const matches = location.hash.match(/filter=([^&]+)/i);
  return matches && decodeURIComponent(matches[1]);
}

// 初始化时应用 Hash 过滤
function initFromHash() {
  const hashFilter = getHashFilter();
  if (hashFilter) {
    grid6.arrange({ filter: hashFilter });
    updateButtons(hashFilter);
  }
}

// Hash 变化时更新过滤
window.addEventListener('hashchange', () => {
  const hashFilter = getHashFilter();
  if (hashFilter) {
    grid6.arrange({ filter: hashFilter });
    updateButtons(hashFilter);
  }
});

// 点击按钮更新 Hash
filterButtons.forEach(btn => {
  btn.addEventListener('click', () => {
    const filterValue = btn.getAttribute('data-filter');
    location.hash = 'filter=' + encodeURIComponent(filterValue);
  });
});

initFromHash();
